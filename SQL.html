Example SQL Queries

-- Example 1: Retrieve all customers from a specific city
SELECT * 
FROM customers 
WHERE city = 'New York';

-- Example 2: Calculate the total revenue by product
SELECT product_id, SUM(price * quantity) AS total_revenue
FROM sales
GROUP BY product_id
ORDER BY total_revenue DESC;

-- Example 3: Find top 5 customers by purchase amount
SELECT customer_id, SUM(price * quantity) AS total_spent
FROM sales
GROUP BY customer_id
ORDER BY total_spent DESC
LIMIT 5;
